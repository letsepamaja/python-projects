# shopping cart
foods = []
prices = []
total = 0
while True:
    food = input('enter food you would like to buy (q to quit): ')
    if food.lower() == 'q':
        break
    else:
        price = float(input(f'enter the price of {food}: R'))
        foods.append(food)
        prices.append(price)
print()
print("********* YOUR CART **********")
print()
for food in foods:
    print(food, end=" ")
print()
for price in prices:
    total = total + price
print()
print(f'the total of your cart is: R{total:02}')
