# manual calculator
numb1 = input('enter the first number: ')
while not numb1.isdigit():
    print('calculator cannot read words ')
    numb1 = input('enter the first number: ')
numb1 = int(numb1)
function = input('enter the function between (*, /, +, -): ')
while function not in ['-', '+', '/', '*']:
    print('please enter the correct function')
    function = input('enter the function between (*, /, +, -): ') 
numb2 = input('enter the second number: ')
while not numb2.isdigit():
    print('calculator cannot read words ')
    numb1 = input('enter the first number: ')
numb2 = int(numb2)
if function == "-":
    result = numb1 - numb2
elif function == '+':
    result = numb1 + numb2
elif function == '/':
    result = numb1 / numb2
elif function == '*':
    result = numb1 * numb2
print(result)
