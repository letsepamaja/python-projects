# countdwn programme
import time

my_time = int(input('enter time in seconds: '))
for encounter in range(my_time, 0, -1):
    seconds = encounter % 60
    minutes = int(encounter / 60) % 60
    hours = int(encounter / 3600) % 60
    days = int(encounter / 86400) % 60
    print(f'{days:02}:{hours:02}:{minutes:02}:{seconds:02}')
    time.sleep(1)
print('time is up!'.upper())
