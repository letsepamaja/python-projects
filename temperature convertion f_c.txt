# temperature converter 	
#(0°C × 1.8) + 32 = 32°F
#(32°F − 32) / 1.8 = 0°C



temperature = input('enter temperature: ')
while not temperature.isdigit():
    print('please enter temperature in numbers')
    temperature = input('enter temperature: ')
temperature = float(temperature)
unit = input('enter the unit of the temperature (f / C): ').lower()

while unit not in ('f', 'c'):
    
    print('please enter a correct unit')
    unit = input('enter the unit of the temperature (f / C): ').lower()


if unit == "f":
    
    result = (temperature - 32) / 1.8
    print(f'your temperature when converted to celcius is: {round(result, 2)}*C')
    
else:
    result = (temperature * 1.8) + 32
    print(f'your temperature when converted to fahrenheit is: {round(result, 2)}*F')
