#banking programme
balance = 0
is_running = True
while is_running:
    print('*********BANKING PROGRAMME*********')
    print('1. Deposit')
    print('2. Withdrawal')
    print('3. Balance')
    print('4. Exit')
    print()
    choice = input('what transaction you would like to make: ')
    print()
    if choice == '1':
        amount = float(input('enter an amount to be deposited: '))
        if  amount > 0:
            balance += amount 
            print('deposited successfully')
        else:
            
            print('that is invalid')
            
        print()
    elif choice == '2':
        amount = float(input('enter an amount to be withdrawn: '))
        if amount <= balance:
            balance -= amount
            print('withdrawn successfully')
        else:
            print('that is invalid')
        
        print()
    elif choice == '3':
        print('the balance is:R', balance)
        
        print()
    elif choice == '4':
        end = input('wanna make another transaction (y,n): ')
        if end == 'n':
            is_running = False
            
    else:
        print('that is invalid')
print('thank you for using this bank')
